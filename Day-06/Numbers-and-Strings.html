lettersMatch2 <- function(words, pattern) {
    list <- grepl(pattern, words)
    return(list)
}

piSeries <- function(n) {
    return(4 * (sum(((1/(((((1:n) - 1) * 2) + 1))) * (-1)^((1:n) - 1)))))
}
piSeries(5)

## [1] 3.34

piSeries(10)

## [1] 3.042

howCloseToPi <- function(n) {
    plot((4 * (cumsum(((1/(((((1:n) - 1) * 2) + 1))) * (-1)^((1:n) - 1))))))
}
